CREATE TABLE TIPO_VIOLAZIONE (
idviolazione INT IDENTITY PRIMARY KEY,

CREATE TABLE VERBALE (
idverbale INT IDENTITY PRIMARY KEY,

FOREIGN KEY (idviolazione) REFERENCES TIPO_VIOLAZIONE(idviolazione)
);
